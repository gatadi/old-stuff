				empStore = new Ext.data.JsonStore({
					root: 'empList',
					autoLoad: true,
					url: url,
					fields: [ {name: 'empId', type: 'int'}, 'firstName', 'lastName', 
					          {name: 'salary', type: 'float' } ],
					sortInfo: { field: 'lastName', direction: 'ASC' }
				}),

					id: 'empGrid',
					store: empStore,
